no1 = int(input("Enter the first number: "))
no2 = int(input("Enter the second number: "))

op = int(input("1.Add  2.Sub  3.Mul  4.Div : "))

if op == 1:
    print("Result:", no1 + no2)

elif op == 2:
    print("Result:", no1 - no2)

elif op == 3:
    print("Result:", no1 * no2)

elif op == 4:
    if no2 != 0:
        print("result", no1 / no2)
        
    else:
        print("cant divide by zero")
        
        
else:
    print("invalid operator")
